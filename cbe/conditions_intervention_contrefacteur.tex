\section{Intervention du contrefacteur présumé}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{longtblr}{colspec = {X[3,l] X[7,l] X[5,l]}, rowsep = 1pt}

\textbf{intervention =} & & \\

\hline
\textbf{personne et \mydots} & \textbf{qui ?} & \\

\ok{peut intervenir}
 & -- tout tiers après qu’une action en contrefaçon fondée sur un brevet EP a été introduite à son encontre
 & \epcart{105}(1)a) \\

 & -- tout tiers après avoir été requis par le titulaire de cesser la contrefaçon et avoir engagé une action en déclaration de non-contrefaçon
 & \epcart{105}(1)b) \\

\nok{ne peut pas intervenir}
 & -- tiers informé d’une réserve sur une requête de cessation de la contrefaçon
 & \epocase{t039297}{T 392/97}, motif 2.3 \\

 & -- tiers après qu’une action en contrefaçon soit fondée sur un brevet d’un État d’extension
 & \epocase{t119608}{T 1196/08}, exergue 2 \\

 & -- opposant n’ayant pas formé recours
 & \epocase{t103800}{T 1038/00} \\

\hline
\textbf{\mydots \delai{délai}} & \textbf{quand ?} & \\

\ok{en opposition et en recours}
 & -- \delai{3 mois à compter de l’action en déclaration de non-contrefaçon}
 & \epcrule{89}(1) \\

 & -- \delai{3 mois à compter de l’action en contrefaçon}
 & \epcrule{89}(1) \\

\ok{en opposition}
 & -- tant qu’une opposition recevable est encore en instance
 & \epcart{105}(1) \\

\ok{en recours}
 & -- tant que le recours est encore en instance
 & \epocase{g000194}{G 1/94}, sommaire \\

 & -- pendant le délai de recours de \delai{2 mois}
 & \epocase{g000491}{G 4/91} \\

 \hline
\textbf{effets juridiques} & \textbf{conséquences ?} & \\

\ok{en opposition}
 & -- l’intervenant acquiert la qualité d’opposant
 & \epcart{105}(2) \\

 & -- l’intervenant peut poursuivre la procédure d’opposition même si les autres opposants retirent leurs oppositions
 & \epocase{g000304}{G 3/04}, motif 10 \\

 & -- l’intervenant peut former recours
 & \epocase{g000304}{G 3/04}, motif 4 \\

\ok{en recours}
 & -- l’intervenant acquiert la qualité d’opposant mais pas celle de requérant
 & \epocase{g000304}{G 3/04}, motif 4 \\

\end{longtblr}
