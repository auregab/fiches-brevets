\section{Nomenclature de demandes}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{longtblr}{colspec = {X[3,l] X[7,l] X[5,l]}, rowsep = 1pt}

\textbf{demande =} & & \\

\hline
\textbf{divisionnaire} & \textbf{délai ?} & \\

\ok{recevable}
 & -- \delai{2 mois à compter de la décision de rejet (délai de recours)}
 & \epocase{g000109}{G 1/09} \\

 & -- \delai{entre le recours sur la délivrance et la délivrance (effet suspensif)}
 & \epocase{j000124}{J 1/24} \\

 & -- \delai{jusqu’à la veille de la publication de la mention de délivrance}
 & \epocase{j002410}{J 24/10} \\

 & \textbf{langue ?} & \\

\ok{recevable}
 & -- langue de procédure
 & \epocase{j001314}{J 13/14} \\

 & \textbf{contenu ?} & \\

\ok{recevable}
 & -- demande
    (i) non conforme à \epcart{123}(2),
    (ii) irrégularité corrigeable
 & \epocase{g000105}{G 1/05}, \epocase{g000106}{G 1/06} \\

 & -- pas le même objet revendiqué qu’un brevet EP délivré pour le même demandeur
 & \epocase{g000419}{G 4/19} \\

 & \textbf{déposant ?} & \\

\nok{irrecevable}
 & -- un des déposants seulement
 & \epocase{j000201}{J 2/01} \\

\hline
\textbf{demande volée} & \textbf{titulaire (présumé usurpateur) ?} & \\

\ok{compétence}
 & -- domicilié dans un État contractant $\Rightarrow$ tribunal de l’État contractant du titulaire
 & A2 PR \\

 & -- non domicilié dans un État contractant et demandeur domicilié dans un État contractant
    $\Rightarrow$ tribunal de l’État contractant du demandeur
 & A3 PR \\

 & -- contrat dans un État contractant pour une invention de salarié
    $\Rightarrow$ tribunal de l’État contractant
 & A4 PR \\

 & -- convention entre parties sur la compétence
    $\Rightarrow$ tribunal de l’État contractant convenu
 & A5 PR \\

 & -- demandeur et titulaire non domiciliés dans un État contractant
    $\Rightarrow$ tribunal allemand
 & A6 PR \\

 & \textbf{effet ?} & \\

 & -- décision du tribunal compétent reconnue par les autres tribunaux
 & A9, A11 PR \\

 & -- redéposer une invention volée en bénéficiant de la date de dépôt initiale
 & \epocase{g000392}{G 3/92} \\

\hline
\textbf{dem. Euro-PCT} & \textbf{langue ?} & \\

\ok{recevable}
 & -- langue de procédure EP = langue de procédure PCT
    lorsque la langue PCT est une langue officielle de l’OEB
 & \epocase{g000108}{G 1/08} \\

 & \textbf{effet ?} & \\

 & -- Euro-PCT considérée comme état de la technique \epcart{54}(3)
    lorsque :
    (i) taxe de dépôt payée,
    (ii) traduction dans une langue officielle
 & \epcrule{165} \\

\end{longtblr}
