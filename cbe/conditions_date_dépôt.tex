%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\section{Obtention d’une date de dépôt}
\footnotetext{conditions nécessaires mais pas suffisantes}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{longtblr}{colspec = {X[3,l] X[7,l] X[5,l]}, rowsep = 1pt}

\textbf{date de dépôt =} & & \\

\hline
\textbf{lieu et \mydots} & \textbf{où?} & \\

\ok{recevable}
 & a) auprès de l’Office européen des brevets
 & \epcart{75}(1)a), \epcrule{35}(1) \\

 & b) auprès d’un office national compétent
 & \epcart{75}(1)b) \\

\hline
\textbf{pièces et\mydots} & \textbf{a minima?} & \\
& a) une indication qu'un brevet EP est demandé; & \epcrule{40}(1)a) & \\
& b) des indications permettant d'identifier ou de prendre contact avec le demandeur; & \epcrule{40}(1)b) & \\
& c) une description ou renvoi à une demande déposée antérieurement. & \epcrule{40}(1)c) & \\

\hline
\textbf{langue} & \textbf{traduction?} & \\

\ok{date de dépôt}
& -- langue officielle de l'OEB: FR, EN, DE & \epcart{14}(2) \\

& -- autre langue avec traduction dans une langue officielle de l’OEB
 & \epcart{14}(2) \\

 & -- description dans une première langue et dessins dans une deuxième langue, sous condition de traduction dans une langue officielle
 & \epocase{t038294}{T 382/94} \\

 & -- seule la description doit être traduite dans une langue officielle, pas les dessins
 & \epocase{j000780}{J 7/80} \\

\nok{pas de date de dépôt}
 & -- première partie de la description dans une première langue et deuxième partie de la description dans une autre langue
 & \epocase{j001896}{J 18/96} \\

\hline
\textbf{effet juridique} & \textbf{parties manquantes?} & \\

\ok{date de dépôt initiale}
 & -- demande sans les parties manquantes
 & \epcrule{56}(4) \\

 & -- demande renvoyant à une demande antérieure, à condition de : (i) revendication de priorité;
    (ii) fourniture d'une copie;
    (iii) divulgation intégrale.
 & \epcrule{56}(3) \\

 & -- demande complétée puis retrait des parties manquantes
 & \epcrule{56}(6) \\

cas particulier
 & -- ajout d’une figure
 & \epocase{j001512}{J 15/12} \\

date corrigée
 & -- demande complétée avec les parties manquantes
 & \epcrule{56}(2) \\

\nok{pas de date de dépôt}
 & -- remplacement d’un texte par un autre
 & \epocase{j002510}{J 25/10} \\

 & -- remplacement de dessins de mauvaise qualité
 & \epocase{j001214}{J 12/14} \\

\textbf{\delai{délai}} & \textbf{quand?} & \\
& -- \delai{2 mois à/c de la réception de la notification de l’OEB} & \epcrule{56}\\

\end{longtblr}
