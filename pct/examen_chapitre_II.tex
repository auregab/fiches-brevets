\textbf{examen chap. II} & & \\
\hline
\textbf{principe et \mydots} & \textbf{pourquoi?} & \\
 & -- accélérer la procédure d'examen des
 demandes de brevet
 & \\

\SetHline{2-3}{}
\textbf{base et \mydots} & \textbf{quoi?} & \\
 & -- demande telle que déposée (description,
 revendications et dessins)
 & \\

& -- demande déjà modifiée après réception
du rapport de recherche internationale
& \pctart{19}\\

\ok{recommandé}
& -- demande modifiée présentées en même temps
que la demande d'examen
& \pctart{34}.2.b)+\pctrule{66}.1)\\
 
\nok{base exclues}
 & -- invention qui n'ont pas fait l'objet d'une
 recherche internationale
 & \\

\SetHline{2-3}{}
\textbf{\mydots critères} & \textbf{lesquels?} & \\
\ok{dans l'examen}
 & -- nouveauté, activité inventive, application industrielle
 & \pctart{33}.2+\pctart{33}.3+\pctart{33}.4\\

\nok{exclus à l'examen}
 & -- objet exclus de la brevetabilité
 & \pctart{34}.4.a.i)+\pctrule{67}\\

 & -- absence de clarté manifeste ou
 revendications contenues dans la
 description pas suffisamment fondées
 & \pctart{34}.4.a.ii)\\

\hline
\textbf{1ere OE chap. II} & \textbf{quoi?} & \\

ISA = IPEA
 & -- reprend l’opinion écrite de l’ISA chapitre I
 & \pctrule{66}.1bis.a) \\

IPEA=EP, ISA$\neq$EP
 & -- nouvelle opinion écrite
 & \pctrule{66}.1bis.b) \\

\textbf{réponse} & \textbf{comment?} & \\
modifications et arguments
 & -- pas de garantie lorsque avant de recevoir 2e l’OE
 ou après le délai de réponse stipulé dans l'opinion
 & \pctrule{66}.3\\
 
communications officieuses
 & -- pas de disposition sur une révision
 officielle d'une opinion d'un examinateur
 sauf conclusion contestée d'absence d'unité
 d'invention
 & \pctrule{66}.6, \pctart{34}.3)+\pctrule{13}+\pctrule{68}.3)\\

\delai{délai}
 & -- délai dans l'opinion écrite, en général
 \delai{2 mois / date de l'opinion écrite}
 & \\

\nok{hors délai}
 & -- IPEA pas obligé de tenir compte de la réponse
