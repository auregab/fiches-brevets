exception

\hline
\textbf{date prio =} & \textbf{quoi?} & \\

PCT $\in$ 1 rev. prio.
 & a) la date de dépôt de la demande antérieure
 & \pctart{2}.xi.a) \\
 
 PCT $\in$ plusieurs rev. prio.
 & b) la date de dépôt de la demande antérieure la plus ancienne
 & \pctart{2}.xi.b) \\

 PCT $\notin$ rev. prio.
 & c) la date du dépôt de la demande PCT
 & \pctart{2}.xi.c) \\
