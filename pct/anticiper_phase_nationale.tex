\subsection{Anticiper les exigences nationales}

\begin{longtblr}{colspec = {X[3,l] X[7,l] X[5,l]}, rowsep = 1pt}

\textbf{exigences nat. =} & & \\
\hline
\textbf{objectif} 
 & \textbf{pourquoi?} & \\
 & -- PCT offre au déposant l'option de remplir les exigences nationales
 de manière anticipée et centralisée sous la forme de \textit{déclarations}
 & \pctrule{4}.17+\pctrule{51bis}\\

\SetHline{2-3}{}
\textbf{\delai{delai} et \mydots} & \textbf{quand?} & \\

\ok{principe}
 & -- \delai{16 mois / date de priorité}
 &\\

\ok{exception}
 & -- après 16 mois et avant la date de publication internationale
 la déclaration est réputée reçue le dernier jour du délai de 16 mois
 & \pctrule{26ter}.1 \\

\SetHline{2-3}{}
\textbf{pièces et \mydots} & \textbf{quoi?} & \\

\SetCell[r=2]{}\ok{déclarations permises}
 & -- identité de l'inventeur
 & \pctrule{51bis}.1.a.i) \\

 & -- droit de demander et d'obtenir un brevet
 & \pctrule{51bis}.1.a.ii) \\

 & -- droit de revendiquer la priorité
 & \pctrule{51bis}.1.a.iii) \\

 & -- qualité d'inventeur lorsque DO = USPTO
 doit être signée par chaque inventeur et tous les inventeurs
 doivent être sur une même déclaration
 & \pctrule{51bis}.1.a.iv)+\pctrule{4}.17.iv) \\

 & -- divulgations non opposables
 & \pctrule{51bis}.1.a.v) \\

\SetHline{2-3}{}
\textbf{\mydots conditions}
 & \textbf{quoi?} & \\

\ok{déclaration prise en compte}
 & -- respecter le libellé standard
 & Instr. 211 à 215 \\

\SetCell[r=2]{valign=h}pièce supplémentaire exigible
 & -- divulgations non opposables: déclaration nouvelle, complétée
 et preuves
 & \pctrule{51bis}.1.a.v) \\

 & -- doute raisonnable sur la véracité de la déclaration
 & \pctrule{51bis}.2) \\

\nok{déclaration pas prise en compte}
 & -- DO qui a notifié une incompatibilité
 & \\

\end{longtblr}
