\section{La publication internationale}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{longtblr}{colspec = {X[3,l] X[7,l] X[5,l]}, rowsep = 1pt}

\textbf{publication =} & & \\

\hline
\textbf{delai et \mydots} & \textbf{quand?} & \\

publication officielle & -- \delai{18 mois à compter de la date de priorité}
 & \pctart{21}+\pctrule{48}\\

publication effective & -- publication hebdomadaire par l’IB, en général le premier jeudi suivant l'expiration des 18 mois
 & \\

prépa. technique & -- \delai{15 jours calendaires avant la date de publication effective}
 & \\

exceptions prépa.technique & -- au nouvel an > 15 jours calendaires avant la date de publication effective
 & \pctart{21}+\pctrule{26}bis+Instr. admin. 406\\

\SetHline{2-3}{}
\textbf{contenu et \mydots} & \textbf{quoi?} & \\

 & (i) une page de couverture: données bibliographiques, titre, abrégé, dessin
 & \pctrule{48}.2.i) \\

 & (ii) texte intégral de la demande internationale telle que déposée ou traduite
 & \pctrule{48}.2.ii) \\

 & (iii) le rapport de recherche internationale
 & \pctrule{48}.2.iii) \\

 & (iv) d'autres pièces, par exemple les modifications apportées aux revendications
   selon l'article 19
 & \pctrule{48}.2.iv) \\

\SetHline{2-3}{}
\textbf{langue et \mydots} & \textbf{comment?} & \\

principe & -- dans l’une des dix langues de publication :
    allemand, anglais, arabe, chinois, coréen,
    espagnol, français, japonais, portugais, russe
 & \pctrule{48}.3 \\

langue de dépôt & (a) dans la langue de dépôt lorsque celle-ci
    est une langue de publication
 & \pctrule{48}.3.a) \\

langue de traduction & (b) dans une langue de traduction remise par le déposant
 aux fins de la recherche internationale lorsque celle-ci est une langue de publication
 & \pctrule{48}.3.b)+\pctrule{12}.3 \\

 & (b') dans une langue de traduction aux fins de la publication
 & \pctrule{48}.3.b)+\pctrule{12}.4 \\

langue anglaise & (c) une traduction anglaise du titre de l'invention, de l'abrégé et du RRI
 lorsque la langue de publication n'est pas l'anglais
 & \pctrule{48}.3.c) \\
 
\hline
 & \textbf{effet?} & \\

 & -- entrée de la demande publiée dans l’état de la technique
 & \pctart{21}+\pctrule{34}.1.b.ii) \\

 & -- protection provisoire :
    attention à fournir une traduction pour produire effet
    dans une législation nationale
 & \pctart{29}+Annexes B1 et B2 GD PCT\\

 & -- accès des tiers au dossier du BI
 & \\

\end{longtblr}
