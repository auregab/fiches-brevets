%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\section{Obtenir une date de dépôt international}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{longtblr}{colspec = {X[3,l] X[7,l] X[5,l]}, rowsep = 1pt}

\textbf{dépôt international =} & & \\

\hline
\textbf{personne et \mydots} & \textbf{qui ?} & \\

en principe
 & -- un déposant ayant :
   (i) un lien (domicile ou nationalité),
   (ii) avec l’un quelconque des États contractants,
   (iii) à la date du dépôt
 & \pctart{11}.1(i) \\

 & -- le déposant peut être une personne physique ou morale
 & \pctrule{18} \\

 & -- pluralité de co-déposants
 & \\

\SetHline{2-3}{}
\textbf{lieu et \mydots} & \textbf{où ?} & \\

RO compétent
 & -- office national de l’État PCT
    où le demandeur est domicilié
 & \pctrule{19}.1(a)(i) \\

 & -- office national de l’État PCT
    dont le demandeur est national
 & \pctrule{19}.1(a)(ii) \\
 & -- RO/IB
 & \pctrule{19}.1(a)(iii) \\

 & -- organisation intergouvernementale
    (ex. office régional)
 & \pctrule{19}.1(b) \\
 & -- office national agissant pour l’État
    dont un demandeur est national ou domicilié
 & \\

\SetHline{2-3}{}
\textbf{langue et \mydots} & \textbf{traduction ?} & \\

principe
 & -- une langue acceptée par le RO
 & \pctrule{12}.1(a) \\

 & -- une seule langue
 & \pctart{3}.4(i) \\

 & -- (i) langue de publication (cf. liste) et
    (ii) langue acceptée par une ISA compétente pour ce RO
 & \pctrule{12}.1(b) \\

\SetHline{2-3}{}
\textbf{\mydots pièces} & \textbf{actes ?} & \\

a minima
 & -- \action{requête comportant :
    (i) l’indication « PCT »,
    (ii) la désignation d’au moins un État contractant,
    (iii) le nom du déposant}
 & \pctart{4}.1 \\

en plus & -- \action{une description}
 & \\

 & -- \action{au moins une revendication}
 & \\

\hline
\textbf{effet juridique} & \textbf{effet ?} & \\

 & -- date de dépôt effective dans chacun des États contractants
 & \pctart{11}.3 \\

 & -- dépôt national régulier au sens de la CUP
    dans chacun des États contractants désignés lors du dépôt
 & \pctart{11}.4 \\

\end{longtblr}
