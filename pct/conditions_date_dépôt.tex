%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\section{Obtenir une date de dépôt international}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{longtblr}{colspec = {X[3,l] X[7,l] X[5,l]}, rowsep = 1pt}

\textbf{dépôt =} & & \\

\hline
\textbf{déposant et \mydots} & \textbf{qui ?} & \\

en principe
 & -- un déposant ayant :
   (i) un lien (domicile ou nationalité),
   (ii) avec l’un quelconque des États contractants,
   (iii) à la date du dépôt
 & \pctart{11}.1(i) \\

 & -- le déposant peut être une personne physique ou morale
 & \pctrule{18} \\

 & -- pluralité de co-déposants
 & \\

\SetHline{2-3}{}
\textbf{lieu et \mydots} & \textbf{où ?} & \\

RO compétent
 & -- office national de l’État PCT
    où le demandeur est domicilié
 & \pctrule{19}.1)a)(i) \\

 & -- office national de l’État PCT
    dont le demandeur est national
 & \pctrule{19}.1)a)(ii) \\
 & -- RO/IB
 & \pctrule{19}.1)a)(iii) \\

 & -- organisation intergouvernementale
    (ex. office régional)
 & \pctrule{19}.1)b) \\
 & -- office national agissant pour l’État
    dont un demandeur est national ou domicilié
 & \\

\SetHline{2-3}{}
\textbf{langue et \mydots} & \textbf{traduction ?} & \\

principe
 & -- une langue acceptée par le RO
 & \pctrule{12}.1)a) \\

 & -- (i) langue de publication (cf. liste) et
    (ii) langue acceptée par une ISA compétente pour ce RO
 & \pctrule{12}.1)b) \\

 & -- une seule langue
 & \pctart{3}.4)i) \\

\SetHline{2-3}{}
\textbf{\mydots pièces} & \textbf{quoi?} & \\

\SetCell[r=3]{valign=h}\ok{requises pour obtenir une date de dépôt}
 & a) indication que le dépôt est à titre de demande internationale
 & \pctart{11}.1)iii)a) \\

 & b) au moins une désignation expresse d'un Etat contractant du PCT
 & \pctart{11}.1)iii)b) \\

 & c) identification du demandeur
 & \pctart{11}.1)iii)c) \\

 & d) description
 & \pctart{11}.1)iii)d) \\

 & e) au moins une revendication
 & \pctart{11}.1)iii)e) \\

\SetCell[r=2]{valign=h}\nok{requises mais pas pour obtenir une date de dépôt}
 & -- requête sous formulaire
 & \pctart{14}, \pctrule{4}, \pctrule{26}, \pctrule{51bis}, \pctrule{90} \\

 & -- abrégé, taxes, signature d’au moins un déposant quel qu’il soit, choix de l’ISA lorsqu’au moins deux ISA sont compétentes
 & \pctart{10}, \pctart{16}, \pctart{32}, \pctrule{19}, \pctrule{35}, \pctrule{45bis}, \pctrule{59} \\

\hline
\textbf{effet juridique} & \textbf{effet ?} & \\

 & -- date de dépôt effective dans chacun des États contractants
 & \pctart{11}.3 \\

 & -- dépôt national régulier au sens de la CUP
    dans chacun des États contractants désignés lors du dépôt
 & \pctart{11}.4 \\

\end{longtblr}
