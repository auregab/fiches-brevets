\begin{longtblr}{colspec = {X[3,l] X[8,l] X[3,l]}}

\hline
\textbf{\mydots 1ère dem.} & \textbf{Principe ?} & \epcart{87}(4) \\

\ok{= 1ère dem.} &
-- vérifier qu’aucune demande antérieure ne remplit tous les critères de A87(1) &
\\

& \textbf{Même déposant ?} & \\

\ok{= 1ère dem.} &
-- demande déposée par B après une demande déposée par A et B &
\epocase{t050788eu1}{T788/05} \\

& -- demande déposée par A après une demande déposée par B &
\epocase{t050005eu1}{T5/05} \\

& \textbf{Même État ?} & \\

\ok{= 1ère dem.} &
-- dépôt dans un État CUP/OMC après un dépôt dans un État hors CUP/OMC &
\\

& \textbf{Même invention ?} & \\

\ok{= 1ère dem.} &
-- généralisation d'exemples (pas 1ère dem. pour les exemples spécifiques) &
\epocase{t111222eu1}{T1222/11}, \epocase{g150001ex1}{G1/15} \\

& -- plage de valeurs plus large (partie nouvelle seulement) &
\epocase{t120282eu1}{T282/12} \\

\nok{$\neq$ 1ère dem.} &
-- demande non provisoire après provisoire &
\\
& -- demande de continuation après dem. US &
\\

\hline

\textbf{Dem. ultérieure} &
\textbf{Même invention ?} &
\epcart{87}(4) \\

\ok{$\Longleftrightarrow$ 1ère dem.} &
-- cf. section \ref{sec:verifier-priorite} &
\epcart{87}(4) \\

& \textbf{Même État ?} &
\epcart{87}(4) \\

\ok{$\Longleftrightarrow$ 1ère dem.} &
-- au moins un État contractant commun désigné &
\epcart{79}, \href{https://www.wipo.int/pct/en/texts/rules/r4.htm}{R4.9a PCT} \\

& \textbf{Demande antérieure retirée / abandonnée ?} &
\epcart{87}(4) \\

\ok{$\Longleftrightarrow$ 1ère dem.} &
-- \delai{apprécié à la date de dépôt de la dem. ultérieure} &
\\

& -- \delai{fiction de retrait avec prise d’effet à expiration du délai non respecté} &
\epocase{g980002ep1}{G4/98} \\

& \textbf{Inspection publique ?} &
\epcart{87}(4) \\

\ok{$\Longleftrightarrow$ 1ère dem.} &
-- apprécié à la date de dépôt de la dem. ultérieure &
\\

\nok{$\nLeftrightarrow$ 1ère dem.} &
-- suite à la notification de la demande FR non encore publiée &
\href{https://www.legifrance.gouv.fr/}{L615-4 CPI} \\

& -- consultation d'un tiers du dossier EP lorsque le demandeur
s'en est prévalu à son encontre &
\epcart{128}(2) \\

& \textbf{Pas de droits subsistants ?} &
\epcart{87}(4) \\

\ok{$\Longleftrightarrow$ 1ère dem.} &
-- apprécié à la date de dépôt de la dem. ultérieure &
\\

& -- subsiste une voie de recours possible pour la dem. antérieure &
\\

\nok{$\nLeftrightarrow$ 1ère dem.} &
-- subsiste le bénéfice de dépôt de la dem. antérieure FR &
\href{https://www.legifrance.gouv.fr}{L612-3 CPI},
\href{https://www.legifrance.gouv.fr}{R612-25 CPI} \\

& -- subsiste la continuation ou continuation in part pour dem. antérieure US &
\href{https://uscode.house.gov/view.xhtml?req=granuleid:USC-prelim-title35-section120}{35 USC 120} \\

& -- subsiste le droit à une divisionnaire EP pour la dem. antérieure EP &
\epcart{76} \\

& -- \delai{rétroactivement lorsqu'une voie de recours est exercée pour la dem. antérieure} &
\\

& \textbf{Pas base de priorité ?} &
\epcart{87}(4) \\

\ok{$\Longleftrightarrow$ 1ère dem.} &
-- \delai{apprécié à la date de dépôt de la demande ultérieure} &
\\

\hline

& \textbf{Conditions cumulatives ?} &
\epcart{87}(4) \\

\ok{$\nLeftrightarrow$ 1ère dem.} &
-- demande supplémentaire après retrait de la demande ultérieure &
Mathely p.590 \\

& -- dépôt d'une demande ultérieure après une demande intermédiaire &
Bodenhausen p.48 \\

\hline

& \textbf{Interdiction ?} & \\

\ok{S’applique} &
-- rev. priorité si XX = État CUP/OMC et YY = EP &
\epcart{87}(4) \\

& -- rev. priorité CUP$\rightarrow$CUP &
\href{https://wipolex.wipo.int/en/text/288688}{Art. 4C(4) CUP} \\

& -- rev. priorité OMC$\rightarrow$OMC sauf CUP &
\href{https://www.wto.org/english/docs_e/legal_e/27-trips_01_e.htm}{ADPIC} \\

\nok{Ne s'applique pas} &
-- objet B si ant. = A+B et ult. = A &
\\

& -- FR3 après FR1$\rightarrow$FR2$\rightarrow$EP, FR1 retiré &
\href{https://www.legifrance.gouv.fr}{L612-3 CPI} \\
